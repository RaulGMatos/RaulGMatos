Assignment 2 Description and Code

Write a Visual Basic program that accepts as input the air temperature and the wind velocity. It will use two formulas to calculate the old and new Windchill factors and the change from the old to new.

  • To accommodate both tables, use the original Windchill table parameters.
  • Place initials in the form's title bar.
  • Do not use any variables, only use object properties.
  • Do not use any TextBoxes, only use labels that look like TextBoxes.
  • When executed, the Windchill form appears in the center of the screen, and the form cannot be resized.
  • The form will not display the usual Minimize, Maximize, and Close buttons so that users must use the Exit button.


Below is the code used to create the program:

Public Class frmMain

    Private Sub cmdInformation_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles cmdInformation.Click
        MessageBox.Show("NOTE: Wind speeds below 5 mph or above 40 mph will have little effect on the Wind-chill factor." & Chr(13) & Chr(13) & "RGM", _
                        "Wind Chill Calculation")
    End Sub

    Private Sub cmdExit_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles cmdExit.Click
        Me.Close()
    End Sub

    Private Sub vsbWindSpeed_ValueChanged(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles vsbWindSpeed.ValueChanged
        lblWindSpeedMPH.Text = vsbWindSpeed.Value
        lblOldWindChillF.Text = CInt(91.4 - (0.474677 - 0.020425 * vsbWindSpeed.Value + 0.303107 * Math.Sqrt(vsbWindSpeed.Value)) _
                                           * (91.4 - (vsbTemperature.Value * -1)))
        lblNewWindChillF.Text = CInt(35.74 + (0.6215 * (vsbTemperature.Value * -1)) - 35.75 * (Math.Pow(vsbWindSpeed.Value, 0.16)) _
                                           + (0.4275 * (vsbTemperature.Value * -1)) * (Math.Pow(vsbWindSpeed.Value, 0.16)))
        lblChangeWindChillF.Text = lblNewWindChillF.Text - lblOldWindChillF.Text
    End Sub

    Private Sub vsbTemperature_ValueChanged(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles vsbTemperature.ValueChanged
        lblTempF.Text = vsbTemperature.Value * -1
        lblOldWindChillF.Text = CInt(91.4 - (0.474677 - 0.020425 * vsbWindSpeed.Value + 0.303107 * Math.Sqrt(vsbWindSpeed.Value)) _
                                           * (91.4 - (vsbTemperature.Value * -1)))
        lblNewWindChillF.Text = CInt(35.74 + (0.6215 * (vsbTemperature.Value * -1)) - 35.75 * (Math.Pow(vsbWindSpeed.Value, 0.16)) _
                                           + (0.4275 * (vsbTemperature.Value * -1)) * (Math.Pow(vsbWindSpeed.Value, 0.16)))
        lblChangeWindChillF.Text = lblNewWindChillF.Text - lblOldWindChillF.Text
    End Sub
End Class
